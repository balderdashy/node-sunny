  6   // Requires.
  7   var http = require('http'),
  8     crypto = require('crypto'),
  9     parse = require('url').parse,
 10     util = require('util'),
 11     utils = require("../../utils"),
 12     BaseAuthentication = require("../../base/authentication").Authentication,
 13     AwsAuthentication;
 14 
 15   /**
 16    * AWS Authentication class.
 17    *
 18    * *See*: http://docs.amazonwebservices.com/AmazonS3/latest/API/
 19    *
 20    * @param  {Object}   options     Options object.
 21    * @config {string}   account     Account name.
 22    * @config {string}   secretKey   Secret key.
 23    * @config {string}   [ssl=false] Use SSL?
 24    * @config {string}   [authUrl]   Authentication URL.
 25    * @config {number}   [timeout]   HTTP timeout in seconds.
 26    * @exports AwsAuthentication as provider.aws.Authentication
 27    * @extends base.Authentication
 28    * @constructor
 29    */
 30   AwsAuthentication = function (options) {
 31     // Patch AWS-specific options.
 32     options.authUrl = options.authUrl || "s3.amazonaws.com";
 33 
 34     // Call superclass.
 35     BaseAuthentication.call(this, options);
 36 
 37     this._CUSTOM_HEADER_PREFIX = "x-amz";
 38     this._CUSTOM_HEADER_RE = /^x-amz-/i;
 39     this._SIGNATURE_ID = "AWS";
 40     this._CONN_CLS = require("./connection").Connection;
 41   };
 42 
 43   util.inherits(AwsAuthentication, BaseAuthentication);
 44 
 45   /**
 46    * Return canonical AMZ headers.
 47    *
 48    * > "x-amz headers are canonicalized by:
 49    * > Lower-case header name
 50    * > Headers sorted by header name
 51    * > The values of headers whose names occur more than once should be white
 52    * > space-trimmed and concatenated with comma separators to be compliant
 53    * > with section 4.2 of RFC 2616.
 54    * > remove any whitespace around the colon in the header
 55    * > remove any newlines ('\n') in continuation lines
 56    * > separate headers by newlines ('\n')"
 57    *
 58    * @see http://s3.amazonaws.com/doc/s3-developer-guide/RESTAuthentication.html
 59    */
 60   AwsAuthentication.prototype._getCanonicalHeaders = function (headers) {
 61     var canonHeaders = [],
 62       header,
 63       value,
 64       customHeaders;
 65 
 66     // Extract all Custom headers.
 67     for (header in headers) {
 68       if (headers.hasOwnProperty(header)) {
 69         if (this._CUSTOM_HEADER_RE.test(header)) {
 70           // Extract value and flatten.
 71           value = headers[header];
 72           if (Array.isArray(value)) {
 73             value = value.join(',');
 74           }
 75 
 76           canonHeaders.push(header.toString().toLowerCase() + ":" + value);
 77         }
 78       }
 79     }
 80 
 81     // Sort and create string.
 82     return canonHeaders.sort().join("\n");
 83   };
 84 
 85   /**
 86    * Return canonical AMZ resource.
 87    *
 88    * > "The resource is the bucket and key (if applicable), separated by a '/'.
 89    * > If the request you are signing is for an ACL or a torrent file, you
 90    * > should include ?acl or ?torrent in the resource part of the canonical
 91    * > string. No other query string parameters should be included, however."
 92    *
 93    * @see http://s3.amazonaws.com/doc/s3-developer-guide/RESTAuthentication.html
 94    */
 95   AwsAuthentication.prototype._getCanonicalResource = function (
 96     resource,
 97     headers
 98   ) {
 99     // Strip off cname bucket, if present.
100     var canonResource = parse(resource, true).pathname,
101       bucketRe = new RegExp("." + this._authUrl + "$"),
102       bucketName;
103 
104     if (bucketRe.test(headers.host)) {
105       bucketName = headers.host.replace(bucketRe, '');
106       canonResource = parse("/" + bucketName + resource, true).pathname;
107     }
108 
109     return canonResource;
110   };
111 
112   /**
113    * Create string to sign.
114    *
115    * > "The string to be signed is formed by appending the REST verb,
116    * > content-md5 value, content-type value, date value, canonicalized x-amz
117    * > headers (see recipe below), and the resource; all separated by newlines.
118    * > (If you cannot set the Date header, use the x-amz-date header as
119    * > described below.)"
120    *
121    * Also:
122    *
123    * > "The content-type and content-md5 values are optional, but if you do
124    * > not include them you must still insert a newline at the point where these
125    * > values would normally be inserted."
126    *
127    * @see http://s3.amazonaws.com/doc/s3-developer-guide/RESTAuthentication.html
128    */
129   AwsAuthentication.prototype._getStringToSign = function (
130     method,
131     path,
132     headers
133   ) {
134     var customHeaders = this._getCanonicalHeaders(headers),
135       customResource = this._getCanonicalResource(path, headers),
136       parts;
137 
138     parts = [
139       method,
140       headers['content-md5'] || '',
141       headers['content-type'] || '',
142       headers['date'] ? headers['date'] : '',
143     ];
144 
145     if (customHeaders) {
146       parts.push(customHeaders);
147     }
148     if (customResource) {
149       parts.push(customResource);
150     }
151 
152     return parts.join("\n");
153   };
154 
155   /**
156    * Create signature.
157    * @private
158    */
159   function getSignature(secretKey, stringToSign) {
160     return crypto
161       .createHmac('sha1', secretKey)
162       .update(stringToSign)
163       .digest('base64');
164   }
165   
166   /**
167    * @see base.Authentication#sign
168    */
169   AwsAuthentication.prototype.sign = function (method, path, headers) {
170     path = path || "/";
171     headers = utils.extend(this._getHeaders(headers));
172     headers['authorization'] = this._SIGNATURE_ID + " " + this._account + ":" +
173       getSignature(this._secretKey,
174                    this._getStringToSign(method, path, headers));
175 
176     return headers;
177   };
178 
179   module.exports.Authentication = AwsAuthentication;
180 }());
