 55    * Readable cloud stream.
 56    *
 57    * *Implements* **[``Readable Stream``][0]** interface.
 58    * [0]: http://nodejs.org/docs/v0.4.9/api/streams.html#readable_Stream
 59    *
 60    * @param  {request.AuthenticatedRawRequest}
 61    *                      request     Request object.
 62    * @param   {Object}    [options]   Options object.
 63    * @config  {Function}  [errorFn]   Function to call with error.
 64    * @config  {Function}  [endFn]     Function to call with results. Called:
 65    *                                  endFn(response).
 66    * @exports ReadStream as stream.ReadStream
 67    * @constructor
 68    */
 69   ReadStream = function (request, options) {
 70     options = options || {};
 71     var self = this;
 72 
 73     // Members.
 74     self._readable = true;
 75     self._request = request;
 76     self._realRequest = request.realRequest;
 77     self._requestDone = false;
 78     self._errorHandled = false;
 79     self._response = null;
 80     self._buf = [];
 81     self._errorFn = options.errorFn || null;
 82     self._endFn = options.endFn || null;
 83 
 84     // Called state
 85     self._endCalled = false;
 86     self._destroyCalled = false;
 87     self._destroySoonCalled = false;
 88 
 89     // Error handling
 90     self._handleErr = function (err, response) {
 91       if (!self._errorHandled) {
 92         self._errorHandled = true;
 93         self._readable = false;
 94         self._requestDone = true;
 95         if (self._errorFn) {
 96           self._errorFn(err, self, response);
 97         } else {
 98           self.emit('error', err);
 99         }
100       }
101     };
102 
103     // Errors: straight pass-through (AuthReq has response).
104     self._request.on('error', function (err, response) {
105       self._handleErr(err, response);
106     });
107 
108     // Event binding.
109     self._realRequest.on('response', function (response) {
110       var encoding = self._request._encoding || null;
111 
112       // Store response and set encoding.
113       self._response = response;
114       if (encoding) {
115         response.setEncoding(encoding);
116       }
117 
118       // Add bindings.
119       response.on('data', function (chunk) {
120         var doData = self.listeners('data').length > 0,
121           isError = response.status && response.status >= 400;
122 
123         if (doData && !isError) {
124           // If we have data listeners and not error, emit the data.
125           self.emit('data', chunk);
126         } else {
127           // Otherwise, accumulate as string.
128           self._buf.push(chunk);
129         }
130       });
131       response.on('end', function () {
132         var results = null,
133           msg,
134           err;
135 
136         // Set state.
137         self._readable = false;
138         self._requestDone = true;
139 
140         if (self._endFn) {
141           results = self._endFn(response);
142         }
143 
144         switch (response.statusCode) {
145         case 200:
146           self.emit('end', results, self._request.getMeta(response));
147           break;
148         default:
149           // Everything unknown is an error.
150           msg = utils.bufToStr(self._buf, encoding, 'utf8');
151           err = new CloudError(msg, { response: response });
152           self._handleErr(err, response);
153         }
154       });
155     });
156   };

481   /**
482    * Write to internal buffer.
483    *
484    * Used to accumulate writes for real one-shot cloud PUT on ``end()``.
485    *
486    * @param {Buffer|string} value     Buffer / string to write.
487    * @param {string='utf8'} encoding  Encoding to use if string value.
488    * @private
489    */
490   WriteStream.prototype._addToBuffer = function (value, encoding) {
491     var self = this,
492       oldBuf,
493       oldBufLength,
494       newBuf,
495       newBufLength,
496       concatBuf;
497 
498     if (value) {
499       // Convert to buffer if not already.
500       if (typeof value === 'string') {
501         encoding = encoding || 'utf8';
502         value = new Buffer(value, encoding);
503       }
504 
505       // Append the buffer.
506       if (value instanceof Buffer) {
507         // Helper variables.
508         oldBuf = self._writeBuf;
509         oldBufLength = oldBuf.length;
510         newBuf = value;
511         newBufLength = value.length;
512 
513         // Copy the buffers.
514         concatBuf = new Buffer(oldBufLength + newBufLength);
515         oldBuf.copy(concatBuf, 0, 0, oldBufLength);
516         newBuf.copy(concatBuf, oldBufLength, 0, newBufLength);
517 
518         // Update the internal buffer.
519         self._writeBuf = concatBuf;
520 
521       } else {
522         throw new Error("Unknown value type: " + (typeof value));
523       }
524     }
525   };

527   /**
528    * Write string/buffer to stream.
529    *
530    * Argument based options:
531    *
532    *  - ``write(string, encoding='utf8', [fd])``
533    *  - ``write(buffer)``
534    *
535    * **Note**: ``fd`` parameter is not currently supported.
536    *
537    * **Note**: AWS/GSFD does not support ``Transfer-Encoding: chunked``, so
538    * we accumulate buffers and write it all in one shot at the ``end()`` call.
539    * In the future, parallel upload might be supported.
540    *
541    * This also means that nothing actually happens on the network until an
542    * ``end()`` is called.
543    *
544    * *Implements* **[``write``][0]**
545    * [0]: http://nodejs.org/docs/v0.4.9/api/streams.html#stream.write
546    *
547    * "Writes string with the given encoding to the stream. Returns true if the
548    * string has been flushed to the kernel buffer. Returns false to indicate
549    * that the kernel buffer is full, and the data will be sent out in the
550    * future. The 'drain' event will indicate when the kernel buffer is empty
551    * again. The encoding defaults to 'utf8'.
552    *
553    * If the optional fd parameter is specified, it is interpreted as an
554    * integral file descriptor to be sent over the stream. This is only
555    * supported for UNIX streams, and is silently ignored otherwise. When
556    * writing a file descriptor in this manner, closing the descriptor before
557    * the stream drains risks sending an invalid (closed) FD."
558    *
559    * ... and ...
560    *
561    * "Same as the above except with a raw buffer."
562    *
563    * @param {Buffer|string} value     Buffer / string to write.
564    * @param {string='utf8'} encoding  Encoding to use if string value.
565    * @return {boolean} If buffer is available (always true).
566    */
567   WriteStream.prototype.write = function (value, encoding) {
568     if (this._writable) {
569       this._addToBuffer(value, encoding);
570     }
571 
572     return this._writable;
573   };

578    *
579    * "Terminates the stream with EOF or FIN."
580    *
581    * Argument based options:
582    *
583    *  - ``end()``
584    *  - ``end(string, encoding)``
585    *  - ``end(buffer)``
586    *
587    * @param {Buffer|string} value     Buffer / string to write.
588    * @param {string='utf8'} encoding  Encoding to use if string value.

590   WriteStream.prototype.end = function (value, encoding) {
591     var self = this,
592       req = self._request,
593       byteLength;
594 
595     if (self._writable && !self._endCalled) {
596       // Set object state.
597       self._endCalled = true;
598       self._writable = false;
599 
600       // Write any values to internal buffer.
601       self._addToBuffer(value, encoding);
602 
603       // Add headers for one-shot request, and actually end() request with
604       // the write buffer.
605       req.setHeader('content-length', self._writeBuf.length);
606       req.end(self._writeBuf);
607     }
