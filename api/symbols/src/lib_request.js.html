 61   /**
 62    * End the request.
 63    *
 64    * Typically starts the async code execution.
 65    *
 66    * *Note*: This function can be called multiple times without bad effect.
 67    * Calling code has the option to call ``end()`` once the request is set
 68    * up, or leave it to the end user.
 69    */
 70   Request.prototype.end = function () {
 71     if (!this._ended) {
 72       this._end.apply(this, arguments);
 73       this._ended = true;
 74     }

119   /**
120    * Authenticated raw request wrapper class.
121    *
122    * @param  {base.Authentication} auth  Authentication object.
123    * @param  {Object}   options     Options object.
124    * @config {string}   [method]    HTTP method (verb).
125    * @config {string}   [path]      HTTP path.
126    * @config {Object}   [params]    HTTP path parameters.
127    * @config {Object}   [headers]   HTTP headers.
128    * @config {Object}   [cloudHeaders] Cloud provider headers to add.
129    * @config {Object}   [metadata]  Cloud metadata to add.
130    * @config {string}   [encoding]  Response encoding.
131    * @extends request.Request
132    * @exports AuthenticatedRawRequest as request.AuthenticatedRawRequest
133    * @constructor
134    */
135   AuthenticatedRawRequest = function (auth, options) {
136     options = options || {};
137     var self = this,
138       method = options.method || "GET",
139       path = options.path || "/",
140       params = options.params || {},
141       urlObj,
142       reqOpts;
143 
144     // Patch path to add in query params.
145     if (Object.keys(params).length > 0) {
146       urlObj = url.parse(path, true);
147       urlObj.query = utils.extend(urlObj.query, params);
148       path = url.format(urlObj);
149     }
150 
151     // Sign the headers, create the request.
152     self._auth = auth;
153     self._method = method;
154     self._encoding = options.encoding || null;
155 
156     // Set headers last (so other object members are created).
157     self._headers = auth.sign(method, path, self._getHeaders(options));
158     self._protocol = auth.ssl ? https : http;
159     self._request = self._protocol.request({
160       host: auth.authUrl(),
161       port: auth.port,
162       path: path,
163       method: method,
164       headers: self._headers
165     });
166   };

221   /**
222    * Return full headers from cloud headers and metadata.
223    *
224    * @param  {Object}   options     Options object.
225    * @config {Object}   [headers]   HTTP headers.
226    * @config {Object}   [cloudHeaders] Cloud provider headers to add.
227    * @config {Object}   [metadata]  Cloud metadata to add.
228    * @returns {Object}              HTTP headers.
229    * @private
230    */
231   AuthenticatedRawRequest.prototype._getHeaders = function (options) {
232     options = options || {};
233     var conn = this._auth.connection,
234       headerPrefix = conn.headerPrefix,
235       metaPrefix = conn.metadataPrefix,
236       rawHeaders = {},
237       headers = options.headers || {},
238       cloudHeaders = options.cloudHeaders || {},
239       metadata = options.metadata || {};
240 
241     // Order is metadata, cloud headers, headers.
242     Object.keys(metadata).forEach(function (header) {
243       rawHeaders[metaPrefix + header.toLowerCase()] = metadata[header];
244     });
245     Object.keys(cloudHeaders).forEach(function (header) {
246       rawHeaders[headerPrefix + header.toLowerCase()] = cloudHeaders[header];
247     });
248     Object.keys(headers).forEach(function (header) {
249       rawHeaders[header.toLowerCase()] = headers[header];
250     });
251 
252     return rawHeaders;
253   };
254 
255   /**
256    * Return separate headers, cloud headers and metadata from headers.
257    *
258    * @param  {HttpResponse} response Response object.
259    * @returns {Object} Object of headers, cloudHeaders, metadata.
260    */
261   AuthenticatedRawRequest.prototype.getMeta = function (response) {
262     var conn = this._auth.connection,
263       headerPrefix = conn.headerPrefix,
264       metaPrefix = conn.metadataPrefix,
265       rawHeaders = response.headers,
266       headers = {},
267       cloudHeaders = {},
268       metadata = {};
269 
270     // First try to get metadata, then cloud headers, then headers.
271     Object.keys(rawHeaders).forEach(function (header) {
272       var key = header.toLowerCase();
273       if (startsWith(key, metaPrefix)) {
274         key = key.substring(metaPrefix.length);
275         metadata[key] = rawHeaders[header];
276       } else if (startsWith(key, headerPrefix)) {
277         key = key.substring(headerPrefix.length);
278         cloudHeaders[key] = rawHeaders[header];
279       } else {
280         headers[key] = rawHeaders[header];
281       }
282     });
283 
284     return {
285       headers: headers,
286       cloudHeaders: cloudHeaders,
287       metadata: metadata
288     };
289   };

302   /**
303    * Authenticated request wrapper class.
304    *
305    * **Note**: Accumulates data for final 'end' event instead of passing
306    * through via typical 'data' events.
307    *
308    * @param  {base.Authentication} auth  Authentication object.
309    * @param  {Object}   options     Options object.
310    * @config {string}   [method]    HTTP method (verb).
311    * @config {string}   [path]      HTTP path.
312    * @config {Object}   [params]    HTTP path parameters.
313    * @config {Object}   [headers]   HTTP headers.
314    * @config {Object}   [cloudHeaders] Cloud provider headers to add.
315    * @config {Object}   [metadata]  Cloud metadata to add.
316    * @config {string}   [encoding]  Response encoding.
317    * @config {Function} [errorFn]   errorFn(err, request, [response])
318    *                                Error handler (stops further emission).
319    * @config {Function} [resultsFn] resultsFn(results, request, [response])
320    *                                Successful results data transform.
321    * @extends request.Request
322    * @exports AuthenticatedRequest as request.AuthenticatedRequest
323    * @constructor
324    */
325   AuthenticatedRequest = function (auth, options) {
326     var self = this;
327 
328     AuthenticatedRawRequest.apply(self, arguments);
329 
330     // Additional members.
331     self._buf = [];
332     self._resultsFn = options.resultsFn || null;
333     self._errorFn = options.errorFn || null;
334     self._handleErr = function (err, response) {
335       if (self._errorFn) {
336         self._errorFn(err, self, response);
337       } else {
338         self.emit('error', err, response);
339       }
340     };
341 
342     // Set up bindings.
343     self._request.on('error', function (err) {
344       self._handleErr(err);
345     });
346     self._request.on('response', function (response) {
347       if (self._encoding) {
348         response.setEncoding(self._encoding);
349       }
350 
351       // Shortcut: If no-data response, just return here.
352       //
353       // **Note**: For some reason, when using HTTPS for both AWS and GSFD,
354       // DELETE blob responses would not have an 'end' event. This avoids
355       // the problem by not even bothering with listening.
356       switch (response.statusCode) {
357       case 200: // Check 200 OK for no bytes.
358         if (response.headers && response.headers['content-length'] === "0") {
359           self.processResults(null, response);
360           return;
361         }
362         break;
363       case 204: // Response has no content.
364         self.processResults(null, response);
365         return;
366       default:
367         // Do nothing - continue processing.
368         break;
369       }
370 
371       // Need more processing, listen to response.
372       response.on('data', function (chunk) {
373         self._buf.push(chunk);
374       });
375       response.on('end', function () {
376         var data = null,
377           getData,
378           msg,
379           err;
380 
381         // Handle the buffer, if we have any.
382         if (self._buf.length > 0) {
383           if (self._encoding) {
384             // If encoding, then join as strings.
385             data = self._buf.join('');
386           } else {
387             // Else, return array of buffers.
388             data = self._buf;
389           }
390         }
391 
392         switch (response.statusCode) {
393         case 200:
394           // processResults emits 'end'.
395           self.processResults(data, response);
396           break;
397         default:
398           // Everything unknown is an error.
399           msg = utils.bufToStr(self._buf, self._encoding, 'utf8');
400           err = new CloudError(msg, { response: response });
401           self._handleErr(err, response);
402         }
403       });
404     });
405   };
406 
407   util.inherits(AuthenticatedRequest, AuthenticatedRawRequest);

409   /**
410    * Process data.
411    *
412    * Also emits '``end``' event on processed data.
413    */
414   AuthenticatedRequest.prototype.processResults = function (data, response) {
415     var self = this,
416       meta = self.getMeta(response),
417       results;
418 
419     results = self._resultsFn ? self._resultsFn(data, self, response) : data;
420 
421     self.emit('end', results, meta);
422   };

424   module.exports.AuthenticatedRequest = AuthenticatedRequest;

454   /** @see request.AuthenticatedXmlRequest#processResults */
455   AuthenticatedXmlRequest.prototype.processResults = function (data, response) {
456     var self = this,
457       meta = self.getMeta(response),
458       parser = new xml2js.Parser();
459 
460     // Parse the XML response to JSON.
461     parser.on('end', function (data) {
462       var results = self._resultsFn
463         ? self._resultsFn(data, self, response)
464         : data;
465       self.emit('end', results, meta);
466     });
467     parser.on('error', function (err) {
468       self.emit('error', err, response);
469     });
470 
471     parser.parseString(data);
472   };
