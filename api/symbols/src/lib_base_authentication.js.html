 27   Authentication = function (options) {
 28     // Argument parsing and validation.
 29     options = options || {};
 30     if (!options.account) { throw new Error("No account name."); }

 32     if (!options.authUrl) { throw new Error("No authentication URL."); }
 33 
 34     // Member variables.
 35     this._account = options.account;
 36     this._secretKey = options.secretKey;

 38     this._port = options.port || (this._ssl ? 443 : 80);

104   /**
105    * Create basic request headers.
106    * @private
107    */
108   Authentication.prototype._getHeaders = function (headers) {
109     var lowHeaders = {},
110       header;
111 
112     // Get lower-cased headers.
113     headers = headers || {};
114     for (header in headers) {
115       if (headers.hasOwnProperty(header)) {
116         lowHeaders[header.toString().toLowerCase()] = headers[header];
117       }
118     }
119 
120     // Add default parameters.
121     lowHeaders['date'] = lowHeaders['date'] || new Date().toUTCString();
122     lowHeaders['host'] = lowHeaders['host'] || this._authUrl;
123 
124     return lowHeaders;
125   };
